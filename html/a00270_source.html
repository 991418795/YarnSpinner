128  void RefreshAllResults() {
129  // Start checking all files.
130 
131  // First, record when we started - we need
132  // to know if it's time to show a progress
133  // dialogue
134  var startTime = EditorApplication.timeSinceStartup;
135 
136  // Have we presented a progress bar?
137  var progressBarVisible = false;
138 
139  // Start checking all files; the delegate will be called
140  // after each file has been checked
141  CheckAllFiles(delegate(int complete, int total) {
142 
143  // How long have we been at this?
144  var timeSinceStart = EditorApplication.timeSinceStartup - startTime;
145 
146  // If longer than 'timeBeforeProgressBar', show the progress bar
147  if (timeSinceStart > timeBeforeProgressBar) {
148 
149  // Figure out how much of the progress bar should be filled
150  var progress = (float)complete / (float)total;
151 
152  // Describe what we're doing
153  var info = string.Format("Checking file {0} of {1}...", complete, total);
154 
155  // Display or update the bar
156  EditorUtility.DisplayProgressBar("Checking Yarn Files", info, progress);
157 
158  // Record that we need to clear this bar
159  progressBarVisible = true;
160  }
161  });
162 
163  // All done. Get rid of the progress bar if needed.
164  if (progressBarVisible) {
165  EditorUtility.ClearProgressBar();
166  }

279  // Finds all .JSON files, and validates them.
280  delegate void GUICallback(int complete, int total);
281  void CheckAllFiles (GUICallback callback = null)
282  {
283 
284  // The shared context for all script analysis.
285  var analysisContext = new Yarn.Analysis.Context();
286 
287  // We shouldn't try to perform program analysis if
288  // any of the files fails to compile, because that
289  // analysis would be performed on incomplete data.
290  bool shouldPerformAnalysis = true;
291 
292  // How many files have we finished checking?
293  int complete = 0;
294 
295  // Let's get started!
296 
297  // First, ensure that we're looking at all of the scripts.
298  UpdateYarnScriptList();
299 
300  // Next, compile each one.
301  foreach (var result in checkResults) {
302 
303  // Attempt to compile the file. Record any compiler messages.
304  CheckerResult.State state;
305 
306  var messages = ValidateFile(result.script, analysisContext, out state);
307 
308  result.state = state;
309  result.messages = messages;
310 
311  // Don't perform whole-program analysis if any file failed to compile
312  if (result.state != CheckerResult.State.Passed) {
313  shouldPerformAnalysis = false;
314  }
315 
316  // We're done with it; if we have a callback to call after
317  // each file is validated, do so.
318  complete++;
319 
320  if (callback != null)
321  callback(complete, checkResults.Count);
322 
323  }
324 
325  var results = new List<Yarn.Analysis.Diagnosis>();
326 
327  if (shouldPerformAnalysis) {
328  var scriptAnalyses = analysisContext.FinishAnalysis ();
329  results.AddRange (scriptAnalyses);
330  }
331 
332 
333  var environmentAnalyses = AnalyseEnvironment ();
334  results.AddRange (environmentAnalyses);
335 
336  diagnoses = results;
337 
338  }

342  // Validates a single script.
343  ValidationMessage[] ValidateFile(TextAsset script, Analysis.Context analysisContext, out CheckerResult.State result) {
344 
345  // The list of messages we got from the compiler.
346  var messageList = new List<ValidationMessage>();
347 
348  // A dummy variable storage; it won't be used, but Dialogue
349  // needs it.
350  var variableStorage = new Yarn.MemoryVariableStore();
351 
352  // The Dialog object is the compiler.
353  var dialog = new Dialogue(variableStorage);
354 
355  // Whether compilation failed or not; this will be
356  // set to true if any error messages are returned.
357  bool failed = false;
358 
359  // Called when we get an error message. Convert this
360  // message into a ValidationMessage and store it;
361  // additionally, mark that this file failed compilation
362  dialog.LogErrorMessage = delegate (string message) {
363  var msg = new ValidationMessage();
364  msg.type = MessageType.Error;
365  msg.message = message;
366  messageList.Add(msg);
367 
368  // any errors means this validation failed
369  failed = true;
370  };
371 
372  // Called when we get an error message. Convert this
373  // message into a ValidationMessage and store it
374  dialog.LogDebugMessage = delegate (string message) {
375  var msg = new ValidationMessage();
376  msg.type = MessageType.Info;
377  msg.message = message;
378  messageList.Add(msg);
379  };
380 
381  // Attempt to compile this script. Any exceptions will result
382  // in an error message
383  try {
384  dialog.LoadString(script.text,script.name);
385  } catch (System.Exception e) {
386  dialog.LogErrorMessage(e.Message);
387  }
388 
389  // Once compilation has finished, run the analysis on it
390  dialog.Analyse(analysisContext);
391 
392  // Did it succeed or not?
393  if (failed) {
394  result = CheckerResult.State.Failed;
395  } else {
396  result = CheckerResult.State.Passed;
397  }
398 
399  // All done.
400  return messageList.ToArray();
401 
402  }

425  IEnumerable<Yarn.Analysis.Diagnosis> AnalyseEnvironment ()
426  {
427 
428  var deprecations = new List<Deprecation>();
429 
430  deprecations.Add(new Deprecation(
431  typeof(Yarn.Unity.VariableStorageBehaviour),
432  "SetNumber",
433  "This method is obsolete, and will not be called in future " +
434  "versions of Yarn Spinner. Use SetValue instead."
435  ));
436 
437  deprecations.Add(new Deprecation(
438  typeof(Yarn.Unity.VariableStorageBehaviour),
439  "GetNumber",
440  "This method is obsolete, and will not be called in future " +
441  "versions of Yarn Spinner. Use GetValue instead."
442  ));
443 
444  var results = new List<Yarn.Analysis.Diagnosis>();
445 
446  var assemblies = System.AppDomain.CurrentDomain.GetAssemblies();
447 
448 
449  foreach (var assembly in assemblies) {
450  foreach (var type in assembly.GetTypes()) {
451 
452  foreach (var deprecation in deprecations) {
453  if (!type.IsSubclassOf (deprecation.type))
454  continue;
455 
456  foreach (var method in type.GetMethods ()) {
457  if (method.Name == deprecation.methodName && method.DeclaringType == type) {
458  var message = "{0} implements the {1} method. {2}";
459  message = string.Format (message, type.Name, deprecation.methodName, deprecation.usageNotes);
460  var diagnosis = new Yarn.Analysis.Diagnosis (message, Yarn.Analysis.Diagnosis.Severity.Warning);
461  results.Add (diagnosis);
462  }
463  }
464 
465  }
466  }
467  }
468  return results;
469  }
