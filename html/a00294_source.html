90  // the preprocessor that cleans up things to make it easier on ANTLR
91  // replaces \r\n with \n
92  // adds in INDENTS and DEDENTS where necessary
93  // replaces \t with four spaces
94  // takes in a string of yarn and returns a string the compiler can then use
95  private struct EmissionTuple
96  {
97  public int depth;
98  public bool emitted;
99  public EmissionTuple(int depth, bool emitted)
100  {
101  this.depth = depth;
102  this.emitted = emitted;
103  }
104  }
105  private string preprocessor(string nodeText)
106  {
107  string processed = null;
108 
109  using (StringReader reader = new StringReader(nodeText))
110  {
111  // a list to hold outputLines once they have been cleaned up
112  List<string> outputLines = new List<string>();
113 
114  // a stack to keep track of how far indented we are
115  // made up of ints and bools
116  // ints track the depth, bool tracks if we emitted an indent token
117  // starts with 0 and false so we can never fall off the end of the stack
118  Stack<EmissionTuple> indents = new Stack<EmissionTuple>();
119  indents.Push(new EmissionTuple(0, false));
120 
121  // a bool to determine if we are in a mode where we need to track indents
122  bool shouldTrackNextIndentation = false;
123 
124  char INDENT = '\a';
125  char DEDENT = '\v';
126  //string INDENT = "{";
127  //string DEDENT = "}";
128 
129  string OPTION = "->";
130 
131  string line;
132  while ((line = reader.ReadLine()) != null)
133  {
134  // replacing \t with 4 spaces
135  string tweakedLine = line.Replace("\t", " ");
136  // stripping of any trailing newlines, will add them back in later
137  tweakedLine = tweakedLine.TrimEnd('\r', '\n');
138 
139  // getting the number of indents on this line
140  int lineIndent = tweakedLine.TakeWhile(Char.IsWhiteSpace).Count();
141 
142  // working out if it is an option (ie does it start with ->)
143  bool isOption = tweakedLine.TrimStart(' ').StartsWith(OPTION);
144 
145  // are we in a state where we need to track indents?
146  var previous = indents.Peek();
147  if (shouldTrackNextIndentation && (lineIndent > previous.depth))
148  {
149  indents.Push(new EmissionTuple(lineIndent, true));
150  // adding an indent to the stream
151  // tries to add it to the end of the previous line where possible
152  if (outputLines.Count == 0)
153  {
154  tweakedLine = INDENT + tweakedLine;
155  }
156  else
157  {
158  outputLines[outputLines.Count - 1] = outputLines[outputLines.Count - 1] + INDENT;
159  }
160 
161  shouldTrackNextIndentation = false;
162  }
163  // have we finished with the current block of statements
164  else if (lineIndent < previous.depth)
165  {
166  while (lineIndent < indents.Peek().depth)
167  {
168  var topLevel = indents.Pop();
169 
170  if (topLevel.emitted)
171  {
172  // adding dedents
173  if (outputLines.Count == 0)
174  {
175  tweakedLine = DEDENT + tweakedLine;
176  }
177  else
178  {
179  outputLines[outputLines.Count - 1] = outputLines[outputLines.Count - 1] + DEDENT;
180  }
181  }
182  }
183  }
184  else
185  {
186  shouldTrackNextIndentation = false;
187  }
188 
189  // do we need to track the indents for the next statement?
190  if (isOption)
191  {
192  shouldTrackNextIndentation = true;
193  if (indents.Peek().depth < lineIndent)
194  {
195  indents.Push(new EmissionTuple(lineIndent, false));
196  }
197  }
198  outputLines.Add(tweakedLine);
199  }
200  // mash it all back together now
201  StringBuilder builder = new StringBuilder();
202  foreach (string outLine in outputLines)
203  {
204  builder.Append(outLine);
205  builder.Append("\n");
206  }
207  processed = builder.ToString();
208  }
209 
210  return processed;
211  }

213  // Given a bunch of raw text, load all nodes that were inside it.

217  public Program Load(string text, Library library, string fileName, Program includeProgram, bool showTokens, bool showParseTree, string onlyConsiderNode, NodeFormat format, bool experimentalMode = false)
218  {
219  if (format == NodeFormat.Unknown)
220  {
221  format = GetFormatFromFileName(fileName);
222  }
223 
224  // currently experimental node can only be used on yarn.txt yarn files and single nodes
225  if (experimentalMode && (format == NodeFormat.Text || format == NodeFormat.SingleNodeText))
226  {
227  // this isn't the greatest...
228  if (format == NodeFormat.SingleNodeText)
229  {
230  // it is just the body
231  // need to add a dummy header and body delimiters
232  StringBuilder builder = new StringBuilder();
233  builder.Append("title:Start\n");
234  builder.Append("---\n");
235  builder.Append(text);
236  builder.Append("\n===\n");
237  text = builder.ToString();
238  }

240  string inputString = preprocessor(text);
241  ICharStream input = CharStreams.fromstring(inputString);
242 
243  YarnSpinnerLexer lexer = new YarnSpinnerLexer(input);
244  CommonTokenStream tokens = new CommonTokenStream(lexer);
245 
246  YarnSpinnerParser parser = new YarnSpinnerParser(tokens);
247  // turning off the normal error listener and using ours
248  parser.RemoveErrorListeners();
249  parser.AddErrorListener(ErrorListener.Instance);
250 
251  IParseTree tree = parser.dialogue();
252  AntlrCompiler antlrcompiler = new AntlrCompiler(library);
253  antlrcompiler.Compile(tree);
254 
255  // merging in the other program if requested
256  if (includeProgram != null)
257  {
258  antlrcompiler.program.Include(includeProgram);
259  }
260 
261  return antlrcompiler.program;
262  }
263  else
264  {
265  // The final parsed nodes that were in the file we were given
266  Dictionary<string, Yarn.Parser.Node> nodes = new Dictionary<string, Parser.Node>();
267 
268  // Load the raw data and get the array of node title-text pairs
269 
270  var nodeInfos = GetNodesFromText(text, format);
271 
272  int nodesLoaded = 0;
273 
274  foreach (NodeInfo nodeInfo in nodeInfos)
275  {
276 
277  if (onlyConsiderNode != null && nodeInfo.title != onlyConsiderNode)
278  continue;
279 
280  // Attempt to parse every node; log if we encounter any errors
281 #if CATCH_EXCEPTIONS
282  try
283  {
284 #endif
285 
286  if (nodes.ContainsKey(nodeInfo.title))
287  {
288  throw new InvalidOperationException("Attempted to load a node called " +
289  nodeInfo.title + ", but a node with that name has already been loaded!");
290  }
291 
292  var lexer = new Lexer();
293  var tokens = lexer.Tokenise(nodeInfo.title, nodeInfo.body);
294 
295  if (showTokens)
296  PrintTokenList(tokens);
297 
298  var node = new Parser(tokens, library).Parse();
299 
300  // If this node is tagged "rawText", then preserve its source
301  if (string.IsNullOrEmpty(nodeInfo.tags) == false &&
302  nodeInfo.tags.Contains("rawText"))
303  {
304  node.source = nodeInfo.body;
305  }
306 
307  node.name = nodeInfo.title;
308 
309  node.nodeTags = nodeInfo.tagsList;
310 
311  if (showParseTree)
312  PrintParseTree(node);
313 
314  nodes[nodeInfo.title] = node;
315 
316  nodesLoaded++;
317 
318 #if CATCH_EXCEPTIONS
319  }
320  catch (Yarn.TokeniserException t)
321  {
322  // Add file information
323  var message = string.Format("In file {0}: Error reading node {1}: {2}", fileName, nodeInfo.title, t.Message);
324  throw new Yarn.TokeniserException(message);
325  }
326  catch (Yarn.ParseException p)
327  {
328  var message = string.Format("In file {0}: Error parsing node {1}: {2}", fileName, nodeInfo.title, p.Message);
329  throw new Yarn.ParseException(message);
330  }
331  catch (InvalidOperationException e)
332  {
333  var message = string.Format("In file {0}: Error reading node {1}: {2}", fileName, nodeInfo.title, e.Message);
334  throw new InvalidOperationException(message);
335  }
336 #endif
337  }
338 
339  var compiler = new Yarn.Compiler(fileName);
340 
341  foreach (var node in nodes)
342  {
343  compiler.CompileNode(node.Value);
344  }
345 
346  if (includeProgram != null)
347  {
348  compiler.program.Include(includeProgram);
349  }
350 
351  return compiler.program;
352  }

413  // Given either Twine, JSON or XML input, return an array
414  // containing info about the nodes in that file
415  internal NodeInfo[] GetNodesFromText(string text, NodeFormat format)
416  {
417  // All the nodes we found in this file
418  var nodes = new List<NodeInfo> ();
419 
420  switch (format)
421  {
422  case NodeFormat.SingleNodeText:
423  // If it starts with a comment, treat it as a single-node file
424  var nodeInfo = new NodeInfo();
425  nodeInfo.title = "Start";
426  nodeInfo.body = text;
427  nodes.Add(nodeInfo);
428  break;
429  case NodeFormat.JSON:
430  // Parse it as JSON
431  try
432  {
433  nodes = JsonConvert.DeserializeObject<List<NodeInfo>>(text);
434  }
435  catch (JsonReaderException e)
436  {
437  dialogue.LogErrorMessage("Error parsing Yarn input: " + e.Message);
438  }
439 
440  break;
441  case NodeFormat.Text:
442 
443  // check for the existence of at least one "---"+newline sentinel, which divides
444  // the headers from the body
445 
446  // we use a regex to match either \r\n or \n line endings
447  if (System.Text.RegularExpressions.Regex.IsMatch(text, "---.?\n") == false) {
448  dialogue.LogErrorMessage("Error parsing input: text appears corrupt (no header sentinel)");
449  break;
450  }
451 
452  var headerRegex = new System.Text.RegularExpressions.Regex("(?<field>.*): *(?<value>.*)");
453 
454  var nodeProperties = typeof(NodeInfo).GetProperties();
455 
456  int lineNumber = 0;
457 
458  using (var reader = new System.IO.StringReader(text))
459  {
460  string line;
461  while ((line = reader.ReadLine()) != null)
462  {
463 
464  // Create a new node
465  NodeInfo node = new NodeInfo();
466 
467  // Read header lines
468  do
469  {
470  lineNumber++;
471 
472  // skip empty lines
473  if (line.Length == 0)
474  {
475  continue;
476  }
477 
478  // Attempt to parse the header
479  var headerMatches = headerRegex.Match(line);
480 
481  if (headerMatches == null)
482  {
483  dialogue.LogErrorMessage(string.Format("Line {0}: Can't parse header '{1}'", lineNumber, line));
484  continue;
485  }
486 
487  var field = headerMatches.Groups["field"].Value;
488  var value = headerMatches.Groups["value"].Value;
489 
490  // Attempt to set the appropriate property using this field
491  foreach (var property in nodeProperties)
492  {
493  if (property.Name != field) {
494  continue;
495  }
496 
497  // skip properties that can't be written to
498  if (property.CanWrite == false)
499  {
500  continue;
501  }
502  try
503  {
504  var propertyType = property.PropertyType;
505  object convertedValue;
506  if (propertyType.IsAssignableFrom(typeof(string)))
507  {
508  convertedValue = value;
509  }
510  else if (propertyType.IsAssignableFrom(typeof(int)))
511  {
512  convertedValue = int.Parse(value);
513  }
514  else if (propertyType.IsAssignableFrom(typeof(NodeInfo.Position)))
515  {
516  var components = value.Split(',');
517 
518  // we expect 2 components: x and y
519  if (components.Length != 2)
520  {
521  throw new FormatException();
522  }
523 
524  var position = new NodeInfo.Position();
525  position.x = int.Parse(components[0]);
526  position.y = int.Parse(components[1]);
527 
528  convertedValue = position;
529  }
530  else {
531  throw new NotSupportedException();
532  }
533  // we need to box this because structs are value types,
534  // so calling SetValue using 'node' would just modify a copy of 'node'
535  object box = node;
536  property.SetValue(box, convertedValue, null);
537  node = (NodeInfo)box;
538  break;
539  }
540  catch (FormatException)
541  {
542  dialogue.LogErrorMessage(string.Format("{0}: Error setting '{1}': invalid value '{2}'", lineNumber, field, value));
543  }
544  catch (NotSupportedException)
545  {
546  dialogue.LogErrorMessage(string.Format("{0}: Error setting '{1}': This property cannot be set", lineNumber, field));
547  }
548  }
549  } while ((line = reader.ReadLine()) != "---");
550 
551  lineNumber++;
552 
553  // We're past the header; read the body
554 
555  var lines = new List<string>();
556 
557  // Read header lines until we hit the end of node sentinel or the end of the file
558  while ((line = reader.ReadLine()) != "===" && line != null)
559  {
560  lineNumber++;
561  lines.Add(line);
562  }
563  // We're done reading the lines! Zip 'em up into a string and
564  // store it in the body
565  node.body = string.Join("\n", lines.ToArray());
566 
567  // And add this node to the list
568  nodes.Add(node);
569 
570  // And now we're ready to move on to the next line!
571 
572  }
573  }
574  break;
575  default:
576  throw new InvalidOperationException("Unknown format " + format.ToString());
577  }
578 
579  // hooray we're done
580  return nodes.ToArray();
581  }
582 
583  }
