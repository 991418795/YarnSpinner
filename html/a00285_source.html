236  object YarnFunctionNodeVisitCount (Value[] parameters)
237  {
238  // Determine the node we're checking
239  string nodeName;
240 
241  if (parameters.Length == 0) {
242  // No parameters? Check the current node
243  nodeName = vm.currentNodeName;
244  } else if (parameters.Length == 1) {
245  // A parameter? Check the named node
246  nodeName = parameters [0].AsString;
247 
248  // Ensure this node exists
249  if (NodeExists (nodeName) == false) {
250  var errorMessage = string.Format (CultureInfo.CurrentCulture, "The node {0} does not " + "exist.", nodeName);
251  LogErrorMessage (errorMessage);
252  return 0;
253  }
254  } else {
255  // We got too many parameters
256  var errorMessage = string.Format (CultureInfo.CurrentCulture, "Incorrect number of parameters to " + "visitCount (expected 0 or 1, got {0})", parameters.Length);
257  LogErrorMessage (errorMessage);
258  return 0;
259  }
260 
261  // Figure out how many times this node was run
262  int visitCount = 0;
263  visitedNodeCount.TryGetValue (nodeName, out visitCount);
264 
265  return visitCount;
266  }

327  public void LoadString(string text, string fileName="<input>", bool showTokens=false, bool showParseTree=false, string onlyConsiderNode=null) {
328 
329  if (LogDebugMessage == null) {
330  throw new YarnException ("LogDebugMessage must be set before loading");
331  }
332 
333  if (LogErrorMessage == null) {
334  throw new YarnException ("LogErrorMessage must be set before loading");
335  }
336 
337  // Try to infer the type
338 
339  NodeFormat format;
340 
341  if (text.StartsWith("[", StringComparison.Ordinal)) {
342  // starts with a [? this is probably a JSON array, in which
343  // case we need to reject this.
344 
345  throw new YarnException("This input appears to be a JSON file. Support for loading JSON was removed in version 1.0; please re-save your document as a Yarn text file.");
346 
347  } else if (text.Contains("---")) {
348  // contains a --- delimiter? probably multi node text
349  format = NodeFormat.Text;
350  } else {
351  // fall back to the single node format
352  format = NodeFormat.SingleNodeText;
353  }
354 
355  program = loader.Load(text, library, fileName, program, showTokens, showParseTree, onlyConsiderNode, format);
356 
357  }

365  public IEnumerable<Yarn.Dialogue.RunnerResult> Run(string startNode = DEFAULT_START) {
366 
367  if (LogDebugMessage == null) {
368  throw new YarnException ("LogDebugMessage must be set before running");
369  }
370 
371  if (LogErrorMessage == null) {
372  throw new YarnException ("LogErrorMessage must be set before running");
373  }
374 
375  if (program == null) {
376  LogErrorMessage ("Dialogue.Run was called, but no program was loaded. Stopping.");
377  yield break;
378  }
379 
380  vm = new VirtualMachine (this, program);
381 
382  RunnerResult latestResult;
383 
384  vm.lineHandler = delegate(LineResult result) {
385  latestResult = result;
386  };
387 
388  vm.commandHandler = delegate(CommandResult result) {
389  // Is it the special custom command "<<stop>>"?
390  if (result is CommandResult && (result as CommandResult).command.text == "stop") {
391  vm.Stop();
392  }
393  latestResult = result;
394  };
395 
396  vm.nodeCompleteHandler = delegate(NodeCompleteResult result) {
397 
398  // get the count if it's there, otherwise it defaults to 0
399  int count = 0;
400  visitedNodeCount.TryGetValue(vm.currentNodeName, out count);
401 
402  visitedNodeCount[vm.currentNodeName] = count + 1;
403  latestResult = result;
404  };
405 
406  vm.optionsHandler = delegate(OptionSetResult result) {
407  latestResult = result;
408  };
409 
410  if (vm.SetNode (startNode) == false) {
411  yield break;
412  }
413 
414  // Run until the program stops, pausing to yield important
415  // results
416  do {
417 
418  latestResult = null;
419  vm.RunNext ();
420  if (latestResult != null)
421  yield return latestResult;
422 
423  } while (vm.executionState != VirtualMachine.ExecutionState.Stopped);
424 
425  }
