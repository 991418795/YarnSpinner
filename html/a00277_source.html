71  // For each default variable that's been defined, parse the string
72  // that the user typed in in Unity and store the variable
73  foreach (var variable in defaultVariables) {
74 
75  object value;
76 
77  switch (variable.type) {
78  case Yarn.Value.Type.Number:
79  float f = 0.0f;
80  float.TryParse(variable.value, out f);
81  value = f;
82  break;
83 
84  case Yarn.Value.Type.String:
85  value = variable.value;
86  break;
87 
88  case Yarn.Value.Type.Bool:
89  bool b = false;
90  bool.TryParse(variable.value, out b);
91  value = b;
92  break;
93 
94  case Yarn.Value.Type.Variable:
95  // We don't support assigning default variables from other variables
96  // yet
97  Debug.LogErrorFormat("Can't set variable {0} to {1}: You can't " +
98  "set a default variable to be another variable, because it " +
99  "may not have been initialised yet.", variable.name, variable.value);
100  continue;
101 
102  case Yarn.Value.Type.Null:
103  value = null;
104  break;
105 
106  default:
107  throw new System.ArgumentOutOfRangeException ();
108 
109  }
110 
111  var v = new Yarn.Value(value);
112 
113  SetValue ("$" + variable.name, v);
114  }
