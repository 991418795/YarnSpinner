209  IEnumerator RunDialogue (string startNode = "Start")
210  {
211  // Mark that we're in conversation.
212  isDialogueRunning = true;
213 
214  // Signal that we're starting up.
215  yield return StartCoroutine(this.dialogueUI.DialogueStarted());
216 
217  // Get lines, options and commands from the Dialogue object,
218  // one at a time.
219  foreach (Yarn.Dialogue.RunnerResult step in dialogue.Run(startNode)) {
220 
221  if (step is Yarn.Dialogue.LineResult) {
222 
223  // Wait for line to finish displaying
224  var lineResult = step as Yarn.Dialogue.LineResult;
225  yield return StartCoroutine (this.dialogueUI.RunLine (lineResult.line));
226 
227  } else if (step is Yarn.Dialogue.OptionSetResult) {
228 
229  // Wait for user to finish picking an option
230  var optionSetResult = step as Yarn.Dialogue.OptionSetResult;
231  yield return StartCoroutine (
232  this.dialogueUI.RunOptions (
233  optionSetResult.options,
234  optionSetResult.setSelectedOptionDelegate
235  ));
236 
237  } else if (step is Yarn.Dialogue.CommandResult) {
238 
239  // Wait for command to finish running
240 
241  var commandResult = step as Yarn.Dialogue.CommandResult;
242 
243  bool hasValidCommand = false;
244  yield return DispatchCommand(commandResult.command.text, (status) => { hasValidCommand = status; });
245  if (!hasValidCommand)
246  {
247  yield return StartCoroutine(this.dialogueUI.RunCommand(commandResult.command));
248  }
249 
250  } else if(step is Yarn.Dialogue.NodeCompleteResult) {
251 
252  // Wait for post-node action
253  var nodeResult = step as Yarn.Dialogue.NodeCompleteResult;
254  yield return StartCoroutine (this.dialogueUI.NodeComplete (nodeResult.nextNode));
255  }
256  }
257 
258  // No more results! The dialogue is done.
259  yield return StartCoroutine (this.dialogueUI.DialogueComplete ());
260 
261  // Clear the 'is running' flag. We do this after DialogueComplete returns,
262  // to allow time for any animations that might run while transitioning
263  // out of a conversation (ie letterboxing going away, etc)
264  isDialogueRunning = false;
265  }

303  public IEnumerator DispatchCommand(string command, System.Action<bool> hasValidCommand) {
304 
305  var words = command.Split(' ');
306 
307  // need 2 parameters in order to have both a command name
308  // and the name of an object to find
309  if (words.Length < 2)
310  {
311  hasValidCommand(false);
312  yield break;
313  }
314 
315  var commandName = words[0];
316 
317  var objectName = words[1];
318 
319  var sceneObject = GameObject.Find(objectName);
320 
321  // If we can't find an object, we can't dispatch a command
322  if (sceneObject == null)
323  {
324  hasValidCommand(false);
325  yield break;
326  }
327 
328  int numberOfMethodsFound = 0;
329  List<string[]> errorValues = new List<string[]>();
330 
331  List<string> parameters;
332 
333  if (words.Length > 2) {
334  parameters = new List<string>(words);
335  parameters.RemoveRange(0, 2);
336  } else {
337  parameters = new List<string>();
338  }
339 
340  // Find every MonoBehaviour (or subclass) on the object
341  foreach (var component in sceneObject.GetComponents<MonoBehaviour>()) {
342  var type = component.GetType();
343 
344  // Find every method in this component
345  foreach (var method in type.GetMethods()) {
346 
347  // Find the YarnCommand attributes on this method
348  var attributes = (YarnCommandAttribute[]) method.GetCustomAttributes(typeof(YarnCommandAttribute), true);
349 
350  // Find the YarnCommand whose commandString is equal to the command name
351  foreach (var attribute in attributes) {
352  if (attribute.commandString == commandName) {
353 
354 
355  var methodParameters = method.GetParameters();
356  bool paramsMatch = false;
357  // Check if this is a params array
358  if (methodParameters.Length == 1 && methodParameters[0].ParameterType.IsAssignableFrom(typeof(string[])))
359  {
360  // Cool, we can send the command!
361  // Yield if this is a Coroutine
362  string[][] paramWrapper = new string[1][];
363  paramWrapper[0] = parameters.ToArray();
364  if (method.ReturnType == typeof(IEnumerator))
365  {
366  yield return StartCoroutine((IEnumerator)method.Invoke(component, paramWrapper));
367  }
368  else
369  {
370  method.Invoke(component, paramWrapper);
371  }
372  numberOfMethodsFound++;
373  paramsMatch = true;
374 
375  }
376  // Otherwise, verify that this method has the right number of parameters
377  else if (methodParameters.Length == parameters.Count)
378  {
379  paramsMatch = true;
380  foreach (var paramInfo in methodParameters)
381  {
382  if (!paramInfo.ParameterType.IsAssignableFrom(typeof(string)))
383  {
384  Debug.LogErrorFormat(sceneObject, "Method \"{0}\" wants to respond to Yarn command \"{1}\", but not all of its parameters are strings!", method.Name, commandName);
385  paramsMatch = false;
386  break;
387  }
388  }
389  if (paramsMatch)
390  {
391  // Cool, we can send the command!
392  // Yield if this is a Coroutine
393  if (method.ReturnType == typeof(IEnumerator))
394  {
395  yield return StartCoroutine((IEnumerator)method.Invoke(component, parameters.ToArray()));
396  }
397  else
398  {
399  method.Invoke(component, parameters.ToArray());
400  }
401  numberOfMethodsFound++;
402  }
403  }
404  //parameters are invalid, but name matches.
405  if (!paramsMatch)
406  {
407  //save this error in case a matching command is never found.
408  errorValues.Add(new string[] { method.Name, commandName, methodParameters.Length.ToString(), parameters.Count.ToString() });
409  }
410  }
411  }
412  }
413  }
414 
415  // Warn if we found multiple things that could respond
416  // to this command.
417  if (numberOfMethodsFound > 1) {
418  Debug.LogWarningFormat(sceneObject, "The command \"{0}\" found {1} targets. " +
419  "You should only have one - check your scripts.", command, numberOfMethodsFound);
420  } else if (numberOfMethodsFound == 0) {
421  //list all of the near-miss methods only if a proper match is not found, but correctly-named methods are.
422  foreach (string[] errorVal in errorValues) {
423  Debug.LogErrorFormat(sceneObject, "Method \"{0}\" wants to respond to Yarn command \"{1}\", but it has a different number of parameters ({2}) to those provided ({3}), or is not a string array!", errorVal[0], errorVal[1], errorVal[2], errorVal[3]);
424  }
425  }
426 
427  hasValidCommand(numberOfMethodsFound > 0);
428  }
429 
430  }
