206  static internal void CheckFileList(IList<string> paths, List<string> allowedExtensions)
207  {
208 
209  if (paths.Count == 0)
210  {
211  Warn("No files provided.");
212  return;
213  }
214 
215  var invalid = new List<string>();
216 
217  foreach (var path in paths)
218  {
219  // Does this file exist?
220  var exists = System.IO.File.Exists(path);
221 
222  // Does this file have the right extension?
223 
224  var hasAllowedExtension = allowedExtensions.FindIndex(item => path.EndsWith(item, StringComparison.InvariantCulture)) != -1;
225 
226  if (!exists || !hasAllowedExtension)
227  {
228  invalid.Add(string.Format(CultureInfo.CurrentCulture, "\"{0}\"", path));
229  }
230  }
231 
232  if (invalid.Count != 0)
233  {
234 
235  var message = string.Format(CultureInfo.CurrentCulture, "The file{0} {1} {2}.",
236  invalid.Count == 1 ? "" : "s",
237  string.Join(", ", invalid.ToArray()),
238  invalid.Count == 1 ? "is not valid" : "are not valid"
239  );
240 
241  Error(message);
242  }

275  static int Run(RunOptions options)
276  {
277 
278  CheckFileList(options.files, ALLOWED_EXTENSIONS);
279 
280  // Create the object that handles callbacks
281  var impl = new ConsoleRunnerImplementation(waitForLines: options.waitForInput);
282 
283  // load the default variables we got on the command line
284  foreach (var variable in options.variables)
285  {
286  var entry = variable.Split(new char[] { '=' }, StringSplitOptions.RemoveEmptyEntries);
287 
288  float value;
289  // If there aren't two parts to this or the second part isn't a float, fail
290  if (entry.Length != 2 || float.TryParse(entry[1], out value) == false)
291  {
292  Warn(string.Format(CultureInfo.CurrentCulture, "Skipping invalid variable {0}", variable));
293  continue;
294  }
295  var name = entry[0];
296 
297  impl.SetNumber(name, value);
298  }
299 
300  if (options.automaticallySelectFirstOption == true)
301  {
302  impl.autoSelectFirstOption = true;
303  }
304 
305  Dialogue dialogue = CreateDialogue(options, impl);
306 
307  // Run the conversation
308 
309  for (int run = 0; run < options.runTimes; run++)
310  {
311  foreach (var step in dialogue.Run(options.startNode))
312  {
313 
314  // It can be one of three types: a line to show, options
315  // to present to the user, or an internal command to run
316 
317  if (step is Dialogue.LineResult)
318  {
319  var lineResult = step as Dialogue.LineResult;
320  impl.RunLine(lineResult.line);
321  }
322  else if (step is Dialogue.OptionSetResult)
323  {
324  var optionsResult = step as Dialogue.OptionSetResult;
325  impl.RunOptions(optionsResult.options, optionsResult.setSelectedOptionDelegate);
326  }
327  else if (step is Dialogue.CommandResult)
328  {
329  var commandResult = step as Dialogue.CommandResult;
330  impl.RunCommand(commandResult.command.text);
331  }
332  }
333  impl.DialogueComplete();
334  }
335  return 0;
336  }

339  {
340 
341  CheckFileList(options.files, ALLOWED_EXTENSIONS);
342 
343  Dialogue dialogue;
344  try
345  {
346  dialogue = CreateDialogue(options, null);
347  }
348 #pragma warning disable CA1031 // Do not catch general exception types
349  catch
350  {
351  return 1;
352  }
353 #pragma warning restore CA1031 // Do not catch general exception types
354 
355  if (options.compileAndExit)
356  {
357  var result = dialogue.GetByteCode();
358  Console.WriteLine(result);
359  return 0;

361 
362  var context = new Yarn.Analysis.Context();
363 
364  dialogue.Analyse(context);
365 
366  foreach (var diagnosis in context.FinishAnalysis())
367  {

373  case Analysis.Diagnosis.Severity.Warning:

375  break;
376  case Analysis.Diagnosis.Severity.Note:
377  Note(diagnosis.ToString(showSeverity: false));
378  break;
379  default:
380  throw new ArgumentOutOfRangeException();
381  }
382 
383  }
384  return 0;

388  static internal Dialogue CreateDialogue(ExecutionOptions options, ConsoleRunnerImplementation impl)
389  {
390 
391  // Load nodes
392  var dialogue = new Dialogue(impl);
393 
394  if (options.experimental)
395  {
396  Warn("Running YarnSpinner in experimental mode may have unexpected behaviour.");
397  dialogue.experimentalMode = true;
398  }
399 
400  // Add some methods for testing
401  dialogue.library.RegisterFunction("add_three_operands", 3, delegate (Value[] parameters)
402  {
403  return parameters[0] + parameters[1] + parameters[2];
404  });
405 
406  dialogue.library.RegisterFunction("last_value", -1, delegate (Value[] parameters)
407  {
408  // return the last value
409  return parameters[parameters.Length - 1];
410  });
411 
412  dialogue.library.RegisterFunction("is_even", 1, delegate (Value[] parameters)
413  {
414  return (int)parameters[0].AsNumber % 2 == 0;
415  });
416 
417  // Register the "assert" function, which stops execution if its parameter evaluates to false
418  dialogue.library.RegisterFunction("assert", -1, delegate (Value[] parameters)
419  {
420  if (parameters[0].AsBool == false)
421  {
422 
423  // TODO: Include file, node and line number
424  if (parameters.Length > 1 && parameters[1].AsBool)
425  {
426  dialogue.LogErrorMessage("ASSERTION FAILED: " + parameters[1].AsString);
427  }
428  else
429  {
430  dialogue.LogErrorMessage("ASSERTION FAILED");
431  }
432  Environment.Exit(1);
433  }
434  });
435 
436 
437  // Register a function to let test scripts register how many
438  // options they expect to send
439  dialogue.library.RegisterFunction("prepare_for_options", 2, delegate (Value[] parameters)
440  {
441  impl.numberOfExpectedOptions = (int)parameters[0].AsNumber;
442  impl.autoSelectOptionNumber = (int)parameters[1].AsNumber;
443  });
444 
445  dialogue.library.RegisterFunction("expect_line", 1, delegate (Value[] parameters)
446  {
447  impl.expectedNextLine = parameters[0].AsString;
448  });
449 
450  dialogue.library.RegisterFunction("expect_command", 1, delegate (Value[] parameters)
451  {
452  impl.expectedNextCommand = parameters[0].AsString;
453  });
454 
455  // If debugging is enabled, log debug messages; otherwise, ignore them
456  if (options.showDebuggingInfo)
457  {
458  dialogue.LogDebugMessage = delegate (string message)
459  {
460  Note(message);
461  };
462  }
463  else
464  {
465  dialogue.LogDebugMessage = delegate (string message) { };
466  }
467 
468  dialogue.LogErrorMessage = delegate (string message)
469  {
470  Warn("Yarn Error: " + message);
471  };
472 
473  foreach (var file in options.files)
474  {
475  try
476  {
477  dialogue.LoadFile(file, false, false, options.onlyConsiderNode);
478  }
479  catch (Yarn.TokeniserException e)
480  {
481  Warn(e.Message);
482  }
483  catch (Yarn.ParseException e)
484  {
485  Warn(e.Message);
486  }
487 
488  }
489 
490  // Load string table
491  if (options.stringTable != null)
492  {
493 
494  var parsedTable = new Dictionary<string, string>();
495 
496  using (var reader = new System.IO.StreamReader(options.stringTable))
497  {
498  using (var csvReader = new CsvReader(reader))
499  {
500  if (csvReader.ReadHeader() == false)
501  {
502  Error(string.Format(CultureInfo.CurrentCulture, "{0} is not a valid string table", options.stringTable));
503  }
504 
505  foreach (var row in csvReader.GetRecords<LocalisedLine>())
506  {
507  parsedTable[row.LineCode] = row.LineText;
508  }
509  }
510  }
511 
512  dialogue.AddStringTable(parsedTable);
513  }
514 
515  return dialogue;
516  }

566  public void RunOptions(Options optionsGroup, OptionChooser optionChooser)
567  {
568 
569  Console.WriteLine("Options:");
570  for (int i = 0; i < optionsGroup.options.Count; i++)
571  {
572  var optionDisplay = string.Format(CultureInfo.CurrentCulture, "{0}. {1}", i + 1, optionsGroup.options[i]);
573  Console.WriteLine(optionDisplay);
574  }
575 
576 
577  // Check to see if the number of expected options
578  // is what we're expecting to see
579  if (numberOfExpectedOptions != -1 &&
580  optionsGroup.options.Count != numberOfExpectedOptions)
581  {
582  // TODO: Output diagnostic info here
583  Error(string.Format(CultureInfo.CurrentCulture, "[ERROR: Expected {0} options, but received {1}]", numberOfExpectedOptions, optionsGroup.options.Count));
584 
585  }
586 
587  // If we were told to automatically select an option, do so
588  if (autoSelectOptionNumber != -1)
589  {
590  Note(string.Format(CultureInfo.CurrentCulture, "[Received {0} options, choosing option {1}]", optionsGroup.options.Count, autoSelectOptionNumber));
591 
592  optionChooser(autoSelectOptionNumber);
593 
594  autoSelectOptionNumber = -1;
595 
596  return;
597 
598  }
599 
600  // Reset the expected options counter
601  numberOfExpectedOptions = -1;
602 
603 
604 
605  if (autoSelectFirstOption == true)
606  {
607  Note("[automatically choosing option 1]");
608  optionChooser(0);
609  return;
610  }
611 
612  do
613  {
614  Console.Write("? ");
615  try
616  {
617  var selectedKey = Console.ReadKey().KeyChar.ToString(CultureInfo.InvariantCulture);
618  int selection;
619 
620  if (int.TryParse(selectedKey, out selection) == true)
621  {
622  Console.WriteLine();
623 
624  // we present the list as 1,2,3, but the API expects
625  // answers as 0,1,2
626  selection -= 1;

629  {
630  Console.WriteLine("Invalid option.");

634  optionChooser(selection);

636  }
637  }
638 
639  }
