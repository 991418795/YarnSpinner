238  // A "text" rule matches everything that it possibly can, up to ANY of
239  // the rules that already exist.
240  public TokenRule AddTextRule (TokenType type, string entersState = null)
241  {
242  if (containsTextRule) {
243  throw new InvalidOperationException ("State already contains a text rule");
244  }
245

247 
248  foreach (var otherRule in tokenRules) {

254  // any of the delimiter rules

468  {
469 
470  // Do some initial setup
471  indentationStack = new Stack<KeyValuePair<int,bool>> ();
472  indentationStack.Push (new KeyValuePair<int, bool>(0, false));
473  shouldTrackNextIndentation = false;
474 
475  var tokens = new TokenList();
476 
477  currentState = defaultState;
478 
479  // Parse each line
480  var lines = new List<string>(text.Split ('\n'));
481  // Add a blank line to ensure that we end with zero indentation
482  lines.Add("");
483 
484  int lineNumber = 1;
485 
486  foreach (var line in lines) {
487  tokens.AddRange (this.TokeniseLine (line, lineNumber));
488  lineNumber++;
489  }
490 
491  var endOfInput = new Token (TokenType.EndOfInput, currentState, lineNumber, 0);
492  tokens.Add (endOfInput);
493 
494  return tokens;

497  TokenList TokeniseLine (string line, int lineNumber)
498  {
499  var lineTokens = new Stack<Token> ();
500 
501  // Replace tabs with four spaces
502  line = line.Replace ("\t", " ");
503 
504  // Strip out \r's
505  line = line.Replace("\r", "");
506 
507  // Record the indentation level if the previous state wants us to
508 
509  var thisIndentation = LineIndentation (line);
510  var previousIndentation = indentationStack.Peek ();
511 
512  if (shouldTrackNextIndentation && thisIndentation > previousIndentation.Key) {
513  // If we are more indented than before, emit an
514  // indent token and record this new indent level
515  indentationStack.Push (new KeyValuePair<int, bool>(thisIndentation, true));
516 
517  var indent = new Token (TokenType.Indent, currentState, lineNumber, previousIndentation.Key);
518  indent.value = "".PadLeft (thisIndentation - previousIndentation.Key);
519 
520  shouldTrackNextIndentation = false;
521 
522  lineTokens.Push (indent);
523 
524  } else if (thisIndentation < previousIndentation.Key) {
525 
526  // If we are less indented, emit a dedent for every
527  // indentation level that we passed on the way back to 0 that also
528  // emitted an indentation token.
529  // at the same time, remove those indent levels from the stack
530 
531  while (thisIndentation < indentationStack.Peek ().Key) {
532 
533  var topLevel = indentationStack.Pop ();
534 
535  if (topLevel.Value) {
536  var dedent = new Token (TokenType.Dedent, currentState, lineNumber, 0);
537  lineTokens.Push (dedent);
538  }
539 
540  }
541  }
542 
543  // Now that we're past any initial indentation, start
544  // finding tokens.
545  int columnNumber = thisIndentation;
546 
547  var whitespace = new Regex (@"\s*");
548 
549  while (columnNumber < line.Length) {
550 
551  // If we're about to hit a line comment, abort processing line
552  // immediately
553  if (line.Substring(columnNumber).StartsWith(LINE_COMMENT, StringComparison.InvariantCulture)) {
554  break;
555  }
556 
557  var matched = false;
558 
559  foreach (var rule in currentState.tokenRules) {
560 
561  var match = rule.regex.Match (line, columnNumber);
562 
563  if (match.Success == false || match.Length == 0)
564  continue;
565 
566  string tokenText;
567 
568  if (rule.type == TokenType.Text) {
569  // if this is text, then back up to the most recent text
570  // delimiting token, and treat everything from there as
571  // the text.
572  // we do this because we don't want this:
573  // <<flip Harley3 +1>>
574  // to get matched as this:
575  // BeginCommand Identifier("flip") Text("Harley3 +1") EndCommand
576  // instead, we want to match it as this:
577  // BeginCommand Text("flip Harley3 +1") EndCommand
578 
579  int textStartIndex = thisIndentation;
580 
581  if (lineTokens.Count > 0) {
582  while (lineTokens.Peek().type == TokenType.Identifier) {
583  lineTokens.Pop ();
584  }
585 
586  var startDelimiterToken = lineTokens.Peek ();
587  textStartIndex = startDelimiterToken.columnNumber;
588  if (startDelimiterToken.type == TokenType.Indent)
589  textStartIndex += startDelimiterToken.value.Length;
590  if (startDelimiterToken.type == TokenType.Dedent)
591  textStartIndex = thisIndentation;
592  }
593 
594  columnNumber = textStartIndex;
595 
596  var textEndIndex = match.Index + match.Length;
597 
598  tokenText = line.Substring (textStartIndex, textEndIndex-textStartIndex);
599 
600  } else {
601  tokenText = match.Value;
602  }
603 
604  columnNumber += tokenText.Length;
605 
606  // If this was a string, lop off the quotes at the start and
607  // end, and un-escape the quotes and slashes
608  if (rule.type == TokenType.String) {
609  tokenText = tokenText.Substring (1, tokenText.Length - 2);
610 
611  tokenText = tokenText.Replace (@"\\", @"\");
612  tokenText = tokenText.Replace (@"\""", @"""");
613  }
614 
615  var token = new Token (rule.type, currentState, lineNumber, columnNumber, tokenText);
616 
617  token.delimitsText = rule.delimitsText;
618 
619  lineTokens.Push (token);
620 
621  if (rule.entersState != null) {
622  if (states.ContainsKey(rule.entersState) == false) {
623  throw new TokeniserException (lineNumber, columnNumber, "Unknown tokeniser state " + rule.entersState);
624  }
625 
626  EnterState (states [rule.entersState]);
627 
628  if (shouldTrackNextIndentation == true) {
629  if (indentationStack.Peek().Key < thisIndentation) {
630  indentationStack.Push (new KeyValuePair<int, bool>(thisIndentation, false));
631  }
632 
633  }
634  }
635 
636  matched = true;
637 
638  break;
639  }
640 
641  if (matched == false) {
642 
643  throw TokeniserException.ExpectedTokensFromState (lineNumber, columnNumber, currentState);
644  }
645 
646  // consume any lingering whitespace before the next token
647  var lastWhitespace = whitespace.Match(line, columnNumber);
648  if (lastWhitespace != null) {
649  columnNumber += lastWhitespace.Length;
650  }
651 
652  }
653 
654  var listToReturn = new TokenList (lineTokens.ToArray ());
655  listToReturn.Reverse ();
656 
657  return listToReturn;
658 
659  }
