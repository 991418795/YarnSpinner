104  public virtual void WriteField( string field )
105  {
106  CheckDisposed();
107 
108  var shouldQuote = configuration.QuoteAllFields;
109 
110  if( configuration.TrimFields )
111  {
112  field = field.Trim();
113  }
114 
115  if( !configuration.QuoteNoFields && !string.IsNullOrEmpty( field ) )
116  {
117  var hasQuote = field.Contains( configuration.QuoteString );
118 
119  if( shouldQuote
120  || hasQuote
121  || field[0] == ' '
122  || field[field.Length - 1] == ' '
123  || field.IndexOfAny( configuration.QuoteRequiredChars ) > -1
124  || ( configuration.Delimiter.Length > 1 && field.Contains( configuration.Delimiter ) ) )
125  {
126  shouldQuote = true;
127  }
128  }
129 
130  WriteField( field, shouldQuote );
131  }

145  public virtual void WriteField( string field, bool shouldQuote )
146  {
147  CheckDisposed();
148 
149  // All quotes must be doubled.
150  if( shouldQuote && !string.IsNullOrEmpty( field ) )
151  {
152  field = field.Replace( configuration.QuoteString, configuration.DoubleQuoteString );
153  }
154 
155  if( configuration.UseExcelLeadingZerosFormatForNumerics && !string.IsNullOrEmpty( field ) && field[0] == '0' && field.All( Char.IsDigit ) )
156  {
157  field = "=" + configuration.Quote + field + configuration.Quote;
158  }
159  else if (shouldQuote)
160  {
161  field = configuration.Quote + field + configuration.Quote;
162  }
163

338  }
339

341  {
342  throw new CsvWriterException( "Configuration.HasHeaderRecord is false. This will need to be enabled to write the header." );
343  }
344 
345  if( hasHeaderBeenWritten )

438  public virtual void WriteRecords( IEnumerable records )
439  {
440  CheckDisposed();
441 
442  Type recordType = null;
443  try
444  {
445  if( configuration.HasExcelSeparator && !hasExcelSeperatorBeenRead )
446  {
447  WriteExcelSeparator();
448  hasExcelSeperatorBeenRead = true;
449  }
450 
451  // Write the header. If records is a List<dynamic>, the header won't be written.
452  // This is because typeof( T ) = Object.
453  var genericEnumerable = records.GetType().GetInterfaces().FirstOrDefault( t => t.GetTypeInfo().IsGenericType && t.GetGenericTypeDefinition() == typeof( IEnumerable<> ) );
454  if( genericEnumerable != null )
455  {
456  recordType = genericEnumerable.GetGenericArguments().Single();
457  var isPrimitive = recordType.GetTypeInfo().IsPrimitive;
458  if( configuration.HasHeaderRecord && !hasHeaderBeenWritten && !isPrimitive )
459  {
460  WriteHeader( recordType );
461  }
462  }
463 
464  foreach( var record in records )
465  {
466  // If records is a List<dynamic>, the header hasn't been written yet.
467  // Write the header based on the record type.
468  recordType = record.GetType();
469  var isPrimitive = recordType.GetTypeInfo().IsPrimitive;
470  if( configuration.HasHeaderRecord && !hasHeaderBeenWritten && !isPrimitive )
471  {
472  WriteHeader( recordType );
473  }
474 
475  try
476  {
477  GetWriteRecordAction( record.GetType() ).DynamicInvoke( record );
478  }
479  catch( TargetInvocationException ex )
480  {
481  throw ex.InnerException;
482  }
483 
484  NextRecord();
485  }
486  }
487  catch( Exception ex )
488  {
489  ExceptionHelper.AddExceptionDataMessage( ex, null, recordType, null, null, null );
490  throw;

563  protected virtual Expression CreatePropertyExpression( Expression recordExpression, CsvClassMap mapping, CsvPropertyMap propertyMap )
564  {
565  if( mapping.PropertyMaps.Any( pm => pm == propertyMap ) )
566  {
567  // The property is on this level.
568  return Expression.Property( recordExpression, propertyMap.Data.Property );
569  }
570 
571  // The property isn't on this level of the mapping.
572  // We need to search down through the reference maps.
573  foreach( var refMap in mapping.ReferenceMaps )
574  {
575  var wrapped = Expression.Property( recordExpression, refMap.Data.Property );
576  var propertyExpression = CreatePropertyExpression( wrapped, refMap.Data.Mapping, propertyMap );
577  if( propertyExpression == null )
578  {
579  continue;
580  }
581 
582  if( refMap.Data.Property.PropertyType.GetTypeInfo().IsValueType )
583  {
584  return propertyExpression;
585  }
586 
587  var nullCheckExpression = Expression.Equal( wrapped, Expression.Constant( null ) );
588 
589  var isValueType = propertyMap.Data.Property.PropertyType.GetTypeInfo().IsValueType;
590  var isGenericType = isValueType && propertyMap.Data.Property.PropertyType.GetTypeInfo().IsGenericType;
591  Type propertyType;
592  if( isValueType && !isGenericType && !configuration.UseNewObjectForNullReferenceProperties )
593  {
594  propertyType = typeof( Nullable<> ).MakeGenericType( propertyMap.Data.Property.PropertyType );
595  propertyExpression = Expression.Convert( propertyExpression, propertyType );
596  }
597  else
598  {
599  propertyType = propertyMap.Data.Property.PropertyType;
600  }
601 
602  var defaultValueExpression = isValueType && !isGenericType
603  ? (Expression)Expression.New( propertyType )
604  : Expression.Constant( null, propertyType );
605  var conditionExpression = Expression.Condition( nullCheckExpression, defaultValueExpression, propertyExpression );
606  return conditionExpression;
607  }
608 
609  return null;
610  }

718  protected virtual void CreateActionForObject( Type type )
719  {
720  var recordParameter = Expression.Parameter( type, "record" );
721 
722  // Get a list of all the properties so they will
723  // be sorted properly.
724  var properties = new CsvPropertyMapCollection();
725  AddProperties( properties, configuration.Maps[type] );
726 
727  if( properties.Count == 0 )
728  {
729  throw new CsvWriterException( string.Format( "No properties are mapped for type '{0}'.", type.FullName ) );
730  }
731 
732  var delegates = new List<Delegate>();
733 
734  foreach( var propertyMap in properties )
735  {
736  if( !CanWrite( propertyMap ) )
737  {
738  continue;
739  }
740 
741  if( propertyMap.Data.TypeConverter == null || !propertyMap.Data.TypeConverter.CanConvertTo( typeof( string ) ) )
742  {
743  // Skip if the type isn't convertible.
744  continue;
745  }
746 
747  var fieldExpression = CreatePropertyExpression( recordParameter, configuration.Maps[type], propertyMap );
748 
749  var typeConverterExpression = Expression.Constant( propertyMap.Data.TypeConverter );
750  if( propertyMap.Data.TypeConverterOptions.CultureInfo == null )
751  {
752  propertyMap.Data.TypeConverterOptions.CultureInfo = configuration.CultureInfo;
753  }
754 
755  var typeConverterOptions = TypeConverterOptions.Merge( TypeConverterOptionsFactory.GetOptions( propertyMap.Data.Property.PropertyType ), propertyMap.Data.TypeConverterOptions );
756  var typeConverterOptionsExpression = Expression.Constant( typeConverterOptions );
757 
758  var method = propertyMap.Data.TypeConverter.GetType().GetMethod( "ConvertToString" );
759  fieldExpression = Expression.Convert( fieldExpression, typeof( object ) );
760  fieldExpression = Expression.Call( typeConverterExpression, method, typeConverterOptionsExpression, fieldExpression );
761 
762  if( type.GetTypeInfo().IsClass )
763  {
764  var areEqualExpression = Expression.Equal( recordParameter, Expression.Constant( null ) );
765  fieldExpression = Expression.Condition( areEqualExpression, Expression.Constant( string.Empty ), fieldExpression );
766  }
767 
768  var writeFieldMethodCall = Expression.Call( Expression.Constant( this ), "WriteField", new[] { typeof( string ) }, fieldExpression );
769 
770  var actionType = typeof( Action<> ).MakeGenericType( type );
771  delegates.Add( Expression.Lambda( actionType, writeFieldMethodCall, recordParameter ).Compile() );
772  }
773 
774  typeActions[type] = CombineDelegates( delegates );
775  }
