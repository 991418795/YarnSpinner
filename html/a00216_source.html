249  protected virtual string[] ReadLine()
250  {
251  string field = null;
252  var fieldStartPosition = readerBufferPosition;
253  var rawFieldStartPosition = readerBufferPosition;
254  var inQuotes = false;
255  var fieldIsEscaped = false;
256  var fieldIsBad = false;
257  var inComment = false;
258  var inDelimiter = false;
259  var inExcelLeadingZerosFormat = false;
260  var delimiterPosition = 0;
261  var prevCharWasDelimiter = false;
262  var recordPosition = 0;
263  record = new string[FieldCount];
264  RawRecord = string.Empty;
265  currentRow++;
266  currentRawRow++;
267 
268  while( true )
269  {
270  if( read )
271  {
272  cPrev = c;
273  }
274 
275  var fieldLength = readerBufferPosition - fieldStartPosition;
276  read = GetChar( out c, ref fieldStartPosition, ref rawFieldStartPosition, ref field, ref fieldIsBad,
277  prevCharWasDelimiter, ref recordPosition, ref fieldLength, inComment, inDelimiter, inQuotes, false );
278  if( !read )
279  {
280  break;
281  }
282  readerBufferPosition++;
283  CharPosition++;
284 
285  #region Use Excel Leading Zeros Format for Numerics
286  // This needs to get in the way and parse things completely different
287  // from how a normal CSV field works. This horribly ugly.
288  if( configuration.UseExcelLeadingZerosFormatForNumerics )
289  {
290  if( c == '=' && !inExcelLeadingZerosFormat && ( prevCharWasDelimiter || cPrev == '\r' || cPrev == '\n' || cPrev == null ) )
291  {
292  // The start of the leading zeros format has been hit.
293 
294  fieldLength = readerBufferPosition - fieldStartPosition;
295  char cNext;
296  GetChar( out cNext, ref fieldStartPosition, ref rawFieldStartPosition, ref field, ref fieldIsBad,
297  prevCharWasDelimiter, ref recordPosition, ref fieldLength, inComment, inDelimiter, inQuotes, true );
298  if( cNext == '"' )
299  {
300  inExcelLeadingZerosFormat = true;
301  continue;
302  }
303  }
304  else if( inExcelLeadingZerosFormat )
305  {
306  if( c == '"' && cPrev == '=' || char.IsDigit( c ) )
307  {
308  // Inside of the field.
309  }
310  else if( c == '"' )
311  {
312  // The end of the field has been hit.
313 
314  char cNext;
315  var peekRead = GetChar( out cNext, ref fieldStartPosition, ref rawFieldStartPosition, ref field, ref fieldIsBad, prevCharWasDelimiter, ref recordPosition, ref fieldLength, inComment, inDelimiter, inQuotes, true );
316  if( cNext == configuration.Delimiter[0] || cNext == '\r' || cNext == '\n' || cNext == '\0' )
317  {
318  AppendField( ref field, fieldStartPosition, readerBufferPosition - fieldStartPosition );
319  UpdateBytePosition( fieldStartPosition, readerBufferPosition - fieldStartPosition );
320  field = field.Trim( '=', '"' );
321  fieldStartPosition = readerBufferPosition;
322 
323  if( !peekRead )
324  {
325  AddFieldToRecord( ref recordPosition, field, ref fieldIsBad );
326  }
327 
328  inExcelLeadingZerosFormat = false;
329  }
330  }
331  else
332  {
333  inExcelLeadingZerosFormat = false;
334  }
335 
336  continue;
337  }
338  }
339  #endregion
340 
341  if( c == configuration.Quote && !configuration.IgnoreQuotes )
342  {
343  if( !fieldIsEscaped && ( prevCharWasDelimiter || cPrev == '\r' || cPrev == '\n' || cPrev == null ) )
344  {
345  // The field is escaped only if the first char of
346  // the field is a quote.
347  fieldIsEscaped = true;
348  }
349 
350  if( !fieldIsEscaped )
351  {
352  fieldIsBad = true;
353 
354  // If the field isn't escaped, the quote
355  // is like any other char and we should
356  // just ignore it.
357  continue;
358  }
359 
360  inQuotes = !inQuotes;
361 
362  if( fieldStartPosition != readerBufferPosition - 1 )
363  {
364  // Grab all the field chars before the
365  // quote if there are any.
366  AppendField( ref field, fieldStartPosition, readerBufferPosition - fieldStartPosition - 1 );
367  // Include the quote in the byte count.
368  UpdateBytePosition( fieldStartPosition, readerBufferPosition - fieldStartPosition );
369  }
370 
371  if( cPrev != configuration.Quote || !inQuotes )
372  {
373  if( inQuotes || cPrev == configuration.Quote || readerBufferPosition == 1 )
374  {
375  // The quote will be uncounted and needs to be catered for if:
376  // 1. It's the opening quote
377  // 2. It's the closing quote on an empty field ("")
378  // 3. It's the closing quote and has appeared as the first character in the buffer
379  UpdateBytePosition( fieldStartPosition, readerBufferPosition - fieldStartPosition );
380  }
381 
382  // Set the new field start position to
383  // the char after the quote.
384  fieldStartPosition = readerBufferPosition;
385  }
386 
387  prevCharWasDelimiter = false;
388 
389  continue;
390  }
391 
392  prevCharWasDelimiter = false;
393 
394  if( fieldIsEscaped && inQuotes )
395  {
396  if( c == '\r' || ( c == '\n' && cPrev != '\r' ) )
397  {
398  currentRawRow++;
399  }
400 
401  // While inside an escaped field,
402  // all chars are ignored.
403  continue;
404  }
405 
406  if( cPrev == configuration.Quote && !configuration.IgnoreQuotes )
407  {
408  if( c != configuration.Delimiter[0] && c != '\r' && c != '\n' )
409  {
410  fieldIsBad = true;
411  }
412 
413  // If we're not in quotes and the
414  // previous char was a quote, the
415  // field is no longer escaped.
416  fieldIsEscaped = false;
417  }
418 
419  if( inComment && c != '\r' && c != '\n' )
420  {
421  // We are on a commented line.
422  // Ignore the character.
423  }
424  else if( c == configuration.Delimiter[0] || inDelimiter )
425  {
426  if( !inDelimiter )
427  {
428  // If we hit the delimiter, we are
429  // done reading the field and can
430  // add it to the record.
431  delimiterPosition = 0;
432  AppendField( ref field, fieldStartPosition, readerBufferPosition - fieldStartPosition - 1 );
433  // Include the comma in the byte count.
434  UpdateBytePosition( fieldStartPosition, readerBufferPosition - fieldStartPosition );
435  AddFieldToRecord( ref recordPosition, field, ref fieldIsBad );
436  fieldStartPosition = readerBufferPosition;
437  field = null;
438 
439  inDelimiter = true;
440  }
441 
442  if( delimiterPosition == configuration.Delimiter.Length - 1 )
443  {
444  // We are done reading the delimeter.
445 
446  // Include the delimiter in the byte count.
447  UpdateBytePosition( fieldStartPosition, readerBufferPosition - fieldStartPosition );
448  inDelimiter = false;
449  prevCharWasDelimiter = true;
450  fieldStartPosition = readerBufferPosition;
451  }
452  else if( configuration.Delimiter[delimiterPosition] != c )
453  {
454  // We're not actually in a delimiter. Reset things back
455  // to the previous field.
456  recordPosition--;
457  fieldStartPosition -= ( delimiterPosition + 1 );
458  inDelimiter = false;
459  }
460  else
461  {
462  delimiterPosition++;
463  }
464  }
465  else if( c == '\r' || c == '\n' )
466  {
467  fieldLength = readerBufferPosition - fieldStartPosition - 1;
468  if( c == '\r' )
469  {
470  char cNext;
471  GetChar( out cNext, ref fieldStartPosition, ref rawFieldStartPosition, ref field, ref fieldIsBad, prevCharWasDelimiter, ref recordPosition, ref fieldLength, inComment, inDelimiter, inQuotes, true );
472  if( cNext == '\n' )
473  {
474  readerBufferPosition++;
475  CharPosition++;
476  }
477  }
478 
479  if( cPrev == '\r' || cPrev == '\n' || inComment || cPrev == null )
480  {
481  // We have hit a blank line. Ignore it.
482 
483  UpdateBytePosition( fieldStartPosition, readerBufferPosition - fieldStartPosition );
484 
485  fieldStartPosition = readerBufferPosition;
486  inComment = false;
487 
488  if( !configuration.IgnoreBlankLines )
489  {
490  break;
491  }
492 
493  // If blank lines are being ignored, we need to add
494  // to the row count because we're skipping the row
495  // and it won't get added normally.
496  currentRow++;
497 
498  continue;
499  }
500 
501  // If we hit the end of the record, add
502  // the current field and return the record.
503  AppendField( ref field, fieldStartPosition, fieldLength );
504  // Include the \r or \n in the byte count.
505  UpdateBytePosition( fieldStartPosition, readerBufferPosition - fieldStartPosition );
506  AddFieldToRecord( ref recordPosition, field, ref fieldIsBad );
507  break;
508  }
509  else if( configuration.AllowComments && c == configuration.Comment && ( cPrev == '\r' || cPrev == '\n' || cPrev == null ) )
510  {
511  inComment = true;
512  }
513  }
514 
515  if( record != null )
516  {
517  RawRecord += new string( readerBuffer, rawFieldStartPosition, readerBufferPosition - rawFieldStartPosition );
518  }
519 
520  return record;
521  }

574  {

576  // rest of the record and field.
577

581  field = "";

589  record = null;
590  }
591

593  return false;
594  }
