153  internal static void AutoMapInternal( CsvClassMap map, bool ignoreReferences, bool prefixReferenceHeaders, LinkedList<Type> mapParents, int indexStart = 0 )
154  {
155  var type = map.GetType().GetTypeInfo().BaseType.GetGenericArguments()[0];
156  if( typeof( IEnumerable ).IsAssignableFrom( type ) )
157  {
158  throw new CsvConfigurationException( "Types that inherit IEnumerable cannot be auto mapped. " +
159  "Did you accidentally call GetRecord or WriteRecord which " +
160  "acts on a single record instead of calling GetRecords or " +
161  "WriteRecords which acts on a list of records?" );
162  }
163 
164  var properties = type.GetProperties( BindingFlags.Instance | BindingFlags.Public );
165  foreach( var property in properties )
166  {
167  var typeConverterType = TypeConverterFactory.GetConverter( property.PropertyType ).GetType();
168  if( typeConverterType == typeof( EnumerableConverter ) )
169  {
170  // The IEnumerable converter just throws an exception so skip it.
171  continue;
172  }
173 
174  var isDefaultConverter = typeConverterType == typeof( DefaultTypeConverter );
175  var hasDefaultConstructor = property.PropertyType.GetConstructor( new Type[0] ) != null;
176  if( isDefaultConverter && hasDefaultConstructor )
177  {
178  if( ignoreReferences )
179  {
180  continue;
181  }
182 
183  // If the type is not one covered by our type converters
184  // and it has a parameterless constructor, create a
185  // reference map for it.
186  if( CheckForCircularReference( property.PropertyType, mapParents ) )
187  {
188  continue;
189  }
190 
191  mapParents.AddLast( type );
192  var refMapType = typeof( DefaultCsvClassMap<> ).MakeGenericType( property.PropertyType );
193  var refMap = (CsvClassMap)ReflectionHelper.CreateInstance( refMapType );
194  AutoMapInternal( refMap, false, prefixReferenceHeaders, mapParents, map.GetMaxIndex() + 1 );
195 
196  if( refMap.PropertyMaps.Count > 0 || refMap.ReferenceMaps.Count > 0 )
197  {
198  var referenceMap = new CsvPropertyReferenceMap( property, refMap );
199  if( prefixReferenceHeaders )
200  {
201  referenceMap.Prefix();
202  }
203 
204  map.ReferenceMaps.Add( referenceMap );
205  }
206  }
207  else
208  {
209  var propertyMap = new CsvPropertyMap( property );
210  propertyMap.Data.Index = map.GetMaxIndex() + 1;
211  if( propertyMap.Data.TypeConverter.CanConvertFrom( typeof( string ) ) ||
212  propertyMap.Data.TypeConverter.CanConvertTo( typeof( string ) ) && !isDefaultConverter )
213  {
214  // Only add the property map if it can be converted later on.
215  // If the property will use the default converter, don't add it because
216  // we don't want the .ToString() value to be used when auto mapping.
217  map.PropertyMaps.Add( propertyMap );
218  }
219  }
220  }
221 
222  map.ReIndex( indexStart );
223  }

232  internal static bool CheckForCircularReference( Type type, LinkedList<Type> mapParents )
233  {
234  if( mapParents.Count == 0 )
235  {
236  return false;
237  }
238 
239  var node = mapParents.Last;
240  while( true )
241  {
242  if( node.Value == type )
243  {
244  return true;
245  }
246 
247  node = node.Previous;
248  if( node == null )
249  {
250  break;
251  }
252  }
253 
254  return false;
255  }
