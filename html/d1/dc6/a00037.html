231  // 1. it has at least 2 words
232  // 2. the second word is the name of an object
233  // 3. that object has components that have methods

238

240  // and the name of an object to find

242  return false;
243 
244  var commandName = words[0];
245 
246  var objectName = words[1];
247 
248  var sceneObject = GameObject.Find(objectName);
249 
250  // If we can't find an object, we can't dispatch a command
251  if (sceneObject == null)
252  return false;
253

255

257

259  parameters = new List<string>(words);
260  parameters.RemoveRange(0, 2);
261  } else {
262  parameters = new List<string>();
263  }
264

266  foreach (var component in sceneObject.GetComponents<MonoBehaviour>()) {
267  var type = component.GetType();
268 
269  // Find every method in this component
270  foreach (var method in type.GetMethods()) {
271 
272  // Find the YarnCommand attributes on this method
273  var attributes = (YarnCommandAttribute[]) method.GetCustomAttributes(typeof(YarnCommandAttribute), true);
274 
275  // Find the YarnCommand whose commandString is equal to the command name
276  foreach (var attribute in attributes) {
277  if (attribute.commandString == commandName) {
278 
279  // Verify that this method has the right number of parameters
280  var methodParameters = method.GetParameters();
281 
282  if (methodParameters.Length != parameters.Count) {
283  Debug.LogErrorFormat(sceneObject, "Method \"{0}\" wants to respond to Yarn command \"{1}\", but it has a different number of parameters ({2}) to those provided ({3})!", method.Name, commandName, methodParameters.Length, parameters.Count);
284  return false;
285  }

288  foreach (var paramInfo in methodParameters) {

290  Debug.LogErrorFormat(sceneObject, "Method \"{0}\" wants to respond to Yarn command \"{1}\", but not all of its parameters are strings!", method.Name, commandName);
291  return false;
292  }
293  }

297  numberOfMethodsFound ++;

299  }
300  }
301  }
302  }
303

305  // to this command.
306  if (numberOfMethodsFound > 1) {
307  Debug.LogWarningFormat(sceneObject, "The command \"{0}\" found {1} targets. " +

309  }
310

312  }
