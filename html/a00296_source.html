207  object YarnFunctionNodeVisitCount (Value[] parameters)
208  {
209  // Determine the node we're checking
210  string nodeName;
211 
212  if (parameters.Length == 0) {
213  // No parameters? Check the current node
214  nodeName = vm.currentNodeName;
215  } else if (parameters.Length == 1) {
216  // A parameter? Check the named node
217  nodeName = parameters [0].AsString;
218 
219  // Ensure this node exists
220  if (NodeExists (nodeName) == false) {
221  var errorMessage = string.Format ("The node {0} does not " + "exist.", nodeName);
222  LogErrorMessage (errorMessage);
223  return 0;
224  }
225  } else {
226  // We got too many parameters
227  var errorMessage = string.Format ("Incorrect number of parameters to " + "visitCount (expected 0 or 1, got {0})", parameters.Length);
228  LogErrorMessage (errorMessage);
229  return 0;
230  }
231 
232  // Figure out how many times this node was run
233  int visitCount = 0;
234  visitedNodeCount.TryGetValue (nodeName, out visitCount);
235 
236  return visitCount;
237  }

344  public void LoadString(string text, string fileName="<input>", bool showTokens=false, bool showParseTree=false, string onlyConsiderNode=null) {
345 
346  if (LogDebugMessage == null) {
347  throw new YarnException ("LogDebugMessage must be set before loading");
348  }
349 
350  if (LogErrorMessage == null) {
351  throw new YarnException ("LogErrorMessage must be set before loading");
352  }
353 
354  // Try to infer the type
355 
356  NodeFormat format;
357 
358  if (text.StartsWith("[", StringComparison.Ordinal)) {
359  // starts with a {? this is probably a JSON array
360  format = NodeFormat.JSON;
361  } else if (text.Contains("---")) {
362  // contains a --- delimiter? probably multi node text
363  format = NodeFormat.Text;
364  } else {
365  // fall back to the single node format
366  format = NodeFormat.SingleNodeText;
367  }
368 
369  program = loader.Load(text, library, fileName, program, showTokens, showParseTree, onlyConsiderNode, format, this.experimentalMode);
370

379  public IEnumerable<Yarn.Dialogue.RunnerResult> Run(string startNode = DEFAULT_START) {
380 
381  if (LogDebugMessage == null) {
382  throw new YarnException ("LogDebugMessage must be set before running");
383  }
384 
385  if (LogErrorMessage == null) {
386  throw new YarnException ("LogErrorMessage must be set before running");
387  }
388 
389  if (program == null) {
390  LogErrorMessage ("Dialogue.Run was called, but no program was loaded. Stopping.");
391  yield break;
392  }
393 
394  vm = new VirtualMachine (this, program);
395 
396  RunnerResult latestResult;
397 
398  vm.lineHandler = delegate(LineResult result) {
399  latestResult = result;
400  };
401 
402  vm.commandHandler = delegate(CommandResult result) {
403  // Is it the special custom command "<<stop>>"?
404  if (result is CommandResult && (result as CommandResult).command.text == "stop") {
405  vm.Stop();
406  }
407  latestResult = result;
408  };
409 
410  vm.nodeCompleteHandler = delegate(NodeCompleteResult result) {
411 
412  // get the count if it's there, otherwise it defaults to 0
413  int count = 0;
414  visitedNodeCount.TryGetValue(vm.currentNodeName, out count);
415 
416  visitedNodeCount[vm.currentNodeName] = count + 1;
417  latestResult = result;
418  };
419 
420  vm.optionsHandler = delegate(OptionSetResult result) {
421  latestResult = result;
422  };
423 
424  if (vm.SetNode (startNode) == false) {
425  yield break;
426  }
427 
428  // Run until the program stops, pausing to yield important
429  // results
430  do {
431 
432  latestResult = null;
433  vm.RunNext ();
434  if (latestResult != null)
435  yield return latestResult;
436 
437  } while (vm.executionState != VirtualMachine.ExecutionState.Stopped);
438 
439  }
